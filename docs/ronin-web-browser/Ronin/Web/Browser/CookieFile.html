def self.save(path,cookies)
  File.open(path,'w') do |file|
    cookies.each do |cookie|
      file.puts(cookie)
    end
  end
end

def each
  return enum_for(__method__) unless block_given?

  File.open(@path) do |file|
    file.each_line(chomp: true) do |line|
      yield Cookie.parse(line)
    end
  end
end
