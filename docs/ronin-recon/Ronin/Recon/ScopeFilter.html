def self.build(values=[])
  host_names = []
  addresses  = []

  values.each do |value|
    case value
    when Values::Host    then host_names << value.name
    when Values::Website then host_names << value.host
    when Values::IP      then addresses  << value.address
    when Values::IPRange then addresses  << value.range
    end
  end

  return new(
    domain_filter: DomainFilter.build(host_names),
    ip_filter:     IPFilter.build(addresses)
  )
end

def ===(value)
  case value
  when Values::Host,
       Values::Domain,
       Values::Nameserver,
       Values::Mailserver
    @domain_filter === value.name
  when Values::Website
    @domain_filter === value.host
  when Values::URL
    values.string
  when Values::IP
    @ip_filter === value.address if @ip_filter
  end
end
