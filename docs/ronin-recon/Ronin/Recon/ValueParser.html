def self.parse(string)
  case string
  when IP_RANGE_REGEX
    Values::IPRange.new(string)
  when IP_REGEX
    Values::IP.new(string)
  when WEBSITE_REGEX
    parse_website(string)
  when WILDCARD_REGEX
    Values::Wildcard.new(string)
  when HOSTNAME_REGEXP
    parse_hostname(string)
  else
    raise(UnknownValue,"unrecognized recon value: #{string.inspect}")
  end
end

def self.parse_hostname(string)
  # remove the public suffix from the host name
  host, _suffix = Support::Network::PublicSuffix.list.split(string)

  if host.include?('.')
    Values::Host.new(string)
  else
    Values::Domain.new(string)
  end
rescue Support::Network::InvalidHostname
  # if the hostname does not have a valid public suffix, assume it's a
  # local hostname.
  Values::Host.new(string)
end

def self.parse_website(string)
  uri = URI.parse(string)

  Values::Website.new(uri.scheme.to_sym,string.host,string.port)
