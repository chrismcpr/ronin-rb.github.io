def initialize(path)
  super(path)

  @files = VALUE_FILE_NAMES.transform_values do |file_name|
    File.open(File.join(@path,file_name),'w')
  end
end

def <<(value)
  file = @files.fetch(value.class) do
    raise(NotImplementedError,"unsupported value class: #{value.inspect}")
  end

  file.puts(value)
  file.flush
end
