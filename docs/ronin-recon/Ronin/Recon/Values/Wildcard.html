def ===(other)
  case other
  when Wildcard
    eql?(other)
  when Domain, Host
    name = other.name

    name.start_with?(@prefix) && name.end_with?(@suffix)
  when URL
    host = other.uri.host

    host.start_with?(@prefix) && host.end_with?(@suffix)
  else
    false
  end
end
