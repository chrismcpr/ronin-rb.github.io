def initialize(workers=DEFAULT_SET)
  @workers = case workers
             when Set   then workers.dup
             when Array then workers.to_set
             else
               raise(ArgumentError,"worker set value must be a Set, Array, or Hash: #{workers.inspect}")
             end
end

def self.from_yaml(value)
  case value
  when Array
    new(value)
  when Hash
    worker_set = new

    value.each do |worker_id,enabled|
      if enabled then worker_set.add(worker_id)
      else            worker_set.delete(worker_id)
      end
    end

    worker_set
  else
    raise(InvalidConfig,"worker_set value must be an Array or a Hash: #{value.inspect}")
  end
end
