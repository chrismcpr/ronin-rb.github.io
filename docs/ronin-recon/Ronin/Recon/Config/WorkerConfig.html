def self.from_yaml(value)
  unless value.kind_of?(Hash)
    raise(InvalidConfig,"worker configuration must be a Hash: #{value.inspect}")
  end

  params      = value[:params] || {}
  concurrency = value[:concurrency]

  unless params.kind_of?(Hash)
    raise(InvalidConfig,"worker params value must be a Hash: #{params.inspect}")
  end

  unless (concurrency.nil? || concurrency.kind_of?(Integer))
    raise(InvalidConfig,"worker concurrency value must be an Integer: #{concurrency.inspect}")
  end

  new(params: params, concurrency: concurrency)
end
