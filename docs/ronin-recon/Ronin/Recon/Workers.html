def intensity(level)
  level_index = INTENSITY_LEVELS.index(level)

  self.class.new(
    @classes.select { |worker|
      if (intensity_index = INTENSITY_LEVELS.index(worker.intensity))
        intensity_index <= level_index
      end
    }
  )
end
