def self.included(command)
  command.option :ip, value: {
                        type:  String,
                        usage: 'IP'
                      },
                      desc: 'Filters the targets by IP'

  command.option :ip_range, value: {
                              type:  String,
                              usage: 'CIDR'
                            },
                            desc: 'Filter the targets by IP range' do |ip_range|
                              options[:ip_range] = IPAddr.new(ip_range)
                            end

  command.option :domain, value: {
                            type:  String,
                            usage: 'DOMAIN',
                          },
                          desc: 'Filters the targets by domain'

  command.option :with_ports, value: {
                                type: /\A(?:\d+|\d+-\d+)(?:,(?:\d+|\d+-\d+))*\z/,
                                usage: '{PORT | PORT1-PORT2},...'
                              },
                              desc: 'Filter targets by port numbers' do |ports|
                                options[:with_ports] = PortList.parse(ports)
                              end

  command.option :with_services, value: {
                                   type: /\A(?:[a-z]+[a-z0-9_+-]*)(?:,[a-z]+[a-z0-9_+-]*)*\z/,
                                   usage: 'SERVICE[,...]'
                                 },
                                 desc: 'Filters targets by service' do |services|
                                   options[:with_services] = Set.new(services.split(','))
                                 end
end

def filter_target_by_port(targets,port_list)
  targets.filter do |host|
    host.each_open_port.any? do |port|
      port_list.include?(port.number)
    end
  end
end

def filter_target_by_service(targets,services)
  targets.filter do |host|
    host.each_open_port.any? do |port|
      if (service = port.service)
        services.include?(service.name)
      end
    end
  end
end

def filter_targets(xml)
  targets = xml.each_up_host.lazy

  if (ip = options[:ip])
    targets = filter_targets_by_ip(targets,ip)
  end

  if (ip_range = options[:ip_range])
    targets = filter_targets_by_ip_range(targets,ip_range)
  end

  if (domain = options[:domain])
    targets = filter_targets_by_domain(targets,domain)
  end

  if (with_ports = options[:with_ports])
    targets = filter_targets_by_port(targets,with_ports)
  end

  return targets
end

def filter_targets_by_domain(targets,domain)
  suffix = ".#{domain}"

  targets.filter do |host|
    if (hostname = host.hostname)
      hostname.name.end_with?(suffix) || hostname.name == domain
    end
  end
end
