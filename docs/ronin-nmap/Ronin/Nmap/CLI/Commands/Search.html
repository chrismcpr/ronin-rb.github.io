def print_target(target)
  if options[:print_ip]
    puts target.address
  elsif options[:print_host]
    puts target.hostname
  elsif options[:print_ip_port]
    target.each_open_port do |port|
      puts "#{target.address}:#{port.number}"
    end
  elsif options[:print_host_port]
    target.each_open_port do |port|
      puts "#{target.hostname}:#{port.number}"
    end
  else
    puts target
  end
end

def run(xml_file)
  unless File.file?(xml_file)
    print_error "no such file or directory: #{xml_file}"
    exit(1)
  end

  xml = ::Nmap::XML.open(xml_file)

  targets = xml.each_host.lazy

  if (ip = options[:ip])
    targets = targets.select do |target|
      target.ip == ip
    end
  end

  if (ip_range = options[:ip_range])
    targets = targets.select do |target|
      ip_range.include?(target.ip)
    end
  end

  if (with_hostname = options[:with_hostname])
    targets = targets.select do |target|
      target.each_hostname.any? do |hostname|
        hostname.name == with_hostname
      end
    end
  end

  if (with_os = options[:with_os])
    targets = targets.select do |target|
      target.os && target.os.each_class.any? do |os_class|
        os_class.family == with_os
      end
    end
  end

  if (with_port = options[:with_port])
    targets = targets.select do |target|
      target.each_open_port.any? do |port|
        port.number == with_port
      end
    end
  end

  if (with_ports = options[:with_ports])
    targets = targets.select do |target|
      target.each_open_port.any? do |port|
        with_ports.include?(port.number)
      end
    end
  end

  if (with_service = options[:with_service])
    targets = targets.select do |target|
      target.each_open_port.any? do |port|
        port.service && port.service.name == with_service
      end
    end
  end

  targets.each do |target|
    print_target(target)
  end
end
