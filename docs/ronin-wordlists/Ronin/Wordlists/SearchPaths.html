def find(name)
  @paths.each do |wordlist_dir|
    if (wordlist_path = wordlist_dir.find(name))
      return wordlist_path
    end
  end

  return nil
end

def list(name='*')

    files.merge(wordlist_dir.list(name))
  end
end

def open(name)
  if (path = find(name))
    Wordlist.open(path)
  else
    raise(WordlistNotFound,"wordlist not found: #{name.inspect}")
  end
