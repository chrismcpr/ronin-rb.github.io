def self.included(command)
  command.option :wordlist_dir, short: '-d',
                                value: {
                                  type: String,
                                  usage: 'DIR'
                                },
                                desc: 'The wordlist directory'
end

def wordlist_dir
  @wordlist_dir ||= if options[:wordlist_dir]
                      WordlistDir.new(options[:wordlist_dir])
                    else
                      CacheDir.new
                    end
end
